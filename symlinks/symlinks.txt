# AutomateLinux Symlinks Reference
# These links are automatically managed by /opt/automateLinux/user_install.sh

# User Configuration
[BASHRC]      ~/.bashrc -> /opt/automateLinux/terminal/bashrc
[TMUX]        ~/.tmux.conf -> /opt/automateLinux/terminal/tmux.conf
[AUTOSTART]   ~/.config/autostart -> /opt/automateLinux/autostart

# GNOME Extensions (~/.local/share/gnome-shell/extensions/)
[EXT_LIB]     lib -> /opt/automateLinux/gnomeExtensions/lib (shared modules for extensions)
[EXT_CLOCK]   clock@ya-niv.com -> /opt/automateLinux/gnomeExtensions/clock@ya-niv.com
[EXT_TRACKER] active-window-tracker@example.com -> /opt/automateLinux/gnomeExtensions/active-window-tracker@example.com
[EXT_WINDOW]  window-selector@ya-niv.com -> /opt/automateLinux/gnomeExtensions/window-selector@ya-niv.com

# System Configuration (Managed by install.sh)
[SERVICE]     /etc/systemd/system/daemon.service -> /opt/automateLinux/daemon/daemon.service (installed via systemd)
[PATH]        /etc/profile.d/automatelinux.sh -> Adds /opt/automateLinux/symlinks to PATH

# Known Issues / Legacy
# [DESKTOP]   Do NOT link /opt/automateLinux/desktop/ to ~/Desktop (causes instability)
